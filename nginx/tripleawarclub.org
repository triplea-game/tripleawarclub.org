server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name tripleawarclub.org www.tripleawarclub.org *.tripleawarclub.org;
    root /usr/share/nginx/html/tripleawarclub.org/public_html;

    ssl_certificate /etc/letsencrypt/live/tripleawarclub.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/tripleawarclub.org/privkey.pem;

    # Turn on OCSP stapling as recommended at
    # https://community.letsencrypt.org/t/integration-guide/13123
    # requires nginx version >= 1.3.7
    ssl_stapling on;
    ssl_stapling_verify on;
    add_header Strict-Transport-Security "max-age=31536000";
    ssl_ciphers 'EECDH+AESGCM:EDH+AESGCM:AES256+EECDH:AES256+EDH';
    ssl_prefer_server_ciphers on;
    ssl_dhparam /etc/dhparam/dhparams.pem;

    access_log /usr/share/nginx/html/tripleawarclub.org/logs/access.log;
    error_log /usr/share/nginx/html/tripleawarclub.org/logs/error.log;

    index  index.html index.htm index.php;
	
    location / {
        try_files $uri $uri/ =404;
    }

    location ~ dice/(keys/)?[^/]+\.(key|dat)$ {
      deny all;
    }

    location ~ .*\.php$ {
        fastcgi_pass unix:/var/run/php/php7.0-fpm.sock;
        include snippets/fastcgi-php.conf;
        include marti_staging_fastcgi_params;
    }
}
